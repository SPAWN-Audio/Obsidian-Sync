# Pseudo Code v2.0

```C#
Initialise Game
	Load UI
	
Reset Game
	Set MaxHealth = 100
	Set PlayerLVL = 1
	Set PlayerXP = 0
	Set PlayerXPThreshold = 100
	Set PlayerHealth = MaxHealth
	Set PlayerAttackStrength = 15
	
While Game is NOT over 
	
	Load Enemy Stats
		Set EnemyLVL = Randomised (between 1-5)
		Set EnemyHealth = MaxHealth
		Set EnemyAttackStrength = 10 + (1.25 * (EnemyLVL - 1))
		Display "A Wild Byte-Chu appeared!"
		
	Player Turn
		EnemyHealth = EnemyHealth - PlayerAttackStrength
		
		IF EnemyHealth is > 0
			Run Enemy Turn
		Else,
			Display: "ENEMY DEFEATED"
			Run XP Reward
	
		XP Reward
			Randomise XP = 10 + (EnemyLVL * RandomNumber(1-5))
			Add XP to PlayerXP
			
			IF PlayerXP >= PlayerXPThreshold
				Display: "Level Up Ready - Press SPACE to evolve"
				
				When SPACE pressed
					PlayerLVL = PlayerLVL + 1
					PlayerAttackStrength = (PlayerAttackStrength + (PlayerLVL * 1.5)
					PlayerHealth = MaxHealth + (PlayerLVL * 1.5)
					PlayerXPThreshold = 100 * PlayerLVL
					
					IF Player LVL = 5
						Display: "Bool-Basaur evolved to Struct-a-saur!"
						Display: "GAME WON"
						END GAME: 
						Break Game Loop
						
			Continue Game Loop
				
	Enemy Turn
		PlayerHealth = PlayerHealth - EnemyAttackStrength
		
		If Player Health > 0
			Go to Player Turn
		Else,
			Display: "GAME OVER"
			END GAME: 
			Break Game Loop

```
